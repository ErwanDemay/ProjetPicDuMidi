247 private function printProgress(string $progress): void
248 {
249 $this->printer->print($progress);
250
251 $this->column++;
252 $this->numberOfTestsRun++;
253
254 if ($this->column === $this->maxColumn || $this->numberOfTestsRun === $this->numberOfTests) {
255 if ($this->numberOfTestsRun === $this->numberOfTests) {
256 $this->printer->print(str_repeat(' ', $this->maxColumn - $this->column));
257 }
258
259 $this->printer->print(
260 sprintf(
261 ' %' . $this->numberOfTestsWidth . 'd / %' .
262 $this->numberOfTestsWidth . 'd (%3s%%)',
263 $this->numberOfTestsRun,
264 $this->numberOfTests,
265 floor(($this->numberOfTestsRun / $this->numberOfTests) * 100)
266 )
267 );
268
269 if ($this->column === $this->maxColumn) {
270 $this->column = 0;
271 $this->printer->print("\n");
272 }
273 }
