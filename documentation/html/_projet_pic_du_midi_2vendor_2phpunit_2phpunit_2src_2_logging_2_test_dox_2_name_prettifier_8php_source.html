168 public function prettifyTestCase(TestCase $test, bool $colorize): string
169 {
170 $annotationWithPlaceholders = false;
171 $methodLevelTestDox = MetadataRegistry::parser()->forMethod($test::class, $test->name())->isTestDox()->isMethodLevel();
172
173 if ($methodLevelTestDox->isNotEmpty()) {
174 $methodLevelTestDox = $methodLevelTestDox->asArray()[0];
175
176 assert($methodLevelTestDox instanceof TestDox);
177
178 $result = $methodLevelTestDox->text();
179
180 if (str_contains($result, '$')) {
181 $annotation = $result;
182 $providedData = $this->mapTestMethodParameterNamesToProvidedDataValues($test, $colorize);
183
184 $variables = array_map(
185 static fn (string $variable): string => sprintf(
186 '/%s(?=\b)/',
187 preg_quote($variable, '/')
188 ),
189 array_keys($providedData)
190 );
191
192 $result = trim(preg_replace($variables, $providedData, $annotation));
193
194 $annotationWithPlaceholders = true;
195 }
196 } else {
197 $result = $this->prettifyTestMethodName($test->name());
198 }
199
200 if (!$annotationWithPlaceholders && $test->usesDataProvider()) {
201 $result .= $this->prettifyDataSet($test, $colorize);
202 }
203

220 private function mapTestMethodParameterNamesToProvidedDataValues(TestCase $test, bool $colorize): array
221 {
222 assert(method_exists($test, $test->name()));
223
225 $reflector = new ReflectionMethod($test::class, $test->name());
226
227 $providedData = [];
228 $providedDataValues = array_values($test->providedData());
229 $i = 0;
230

232
233 foreach ($reflector->getParameters() as $parameter) {
234 if (!array_key_exists($i, $providedDataValues) && $parameter->isDefaultValueAvailable()) {
235 $providedDataValues[$i] = $parameter->getDefaultValue();
236 }
237
238 $value = $providedDataValues[$i++] ?? null;
239
240 if (is_object($value)) {
241 $reflector = new ReflectionObject($value);
242
243 if ($reflector->isEnum()) {
244 $value = $value->value;
245 } elseif ($reflector->hasMethod('__toString')) {
246 $value = (string) $value;

248 $value = $value::class;
249 }
250 }

252 if (!is_scalar($value)) {
253 $value = gettype($value);
254 }
255
256 if (is_bool($value) || is_int($value) || is_float($value)) {
257 $value = (new Exporter)->export($value);
258 }
259
