65 public static function fromXdebugWithPathCoverage(array $rawCoverage): self
66 {
67 $lineCoverage = [];
68 $functionCoverage = [];
69
70 foreach ($rawCoverage as $file => $fileCoverageData) {
71 // Xdebug annotates the function name of traits, strip that off
72 foreach ($fileCoverageData['functions'] as $existingKey => $data) {
73 if (str_ends_with($existingKey, '}') && !str_starts_with($existingKey, '{')) { // don't want to catch {main}
74 $newKey = preg_replace('/\{.*}$/', '', $existingKey);
75 $fileCoverageData['functions'][$newKey] = $data;
76 unset($fileCoverageData['functions'][$existingKey]);
77 }
78 }
79
80 $lineCoverage[$file] = $fileCoverageData['lines'];
81 $functionCoverage[$file] = $fileCoverageData['functions'];
82 }
83
84 return new self($lineCoverage, $functionCoverage);

154 public function markExecutableLineByBranch(string $filename, array $linesToBranchMap): void
155 {
156 if (!isset($this->lineCoverage[$filename])) {
157 return;
158 }
159
160 $linesByBranch = [];
161
162 foreach ($linesToBranchMap as $line => $branch) {
163 $linesByBranch[$branch][] = $line;
164 }
165
166 foreach ($this->lineCoverage[$filename] as $line => $lineStatus) {
167 if (!isset($linesToBranchMap[$line])) {
168 continue;
169 }
170
171 $branch = $linesToBranchMap[$line];
172
173 if (!isset($linesByBranch[$branch])) {
174 continue;
175 }
176
177 foreach ($linesByBranch[$branch] as $lineInBranch) {
178 $this->lineCoverage[$filename][$lineInBranch] = $lineStatus;
179 }
180
181 if (Driver::LINE_EXECUTED === $lineStatus) {
182 unset($linesByBranch[$branch]);
183 }
184 }

190 public function keepFunctionCoverageDataOnlyForLines(string $filename, array $lines): void
191 {
192 if (!isset($this->functionCoverage[$filename])) {
193 return;
194 }
195
196 foreach ($this->functionCoverage[$filename] as $functionName => $functionData) {
197 foreach ($functionData['branches'] as $branchId => $branch) {
198 if (count(array_diff(range($branch['line_start'], $branch['line_end']), $lines)) > 0) {
199 unset($this->functionCoverage[$filename][$functionName]['branches'][$branchId]);
200
201 foreach ($functionData['paths'] as $pathId => $path) {
202 if (in_array($branchId, $path['path'], true)) {
203 unset($this->functionCoverage[$filename][$functionName]['paths'][$pathId]);
204 }
205 }
206 }

233 unset($this->functionCoverage[$filename][$functionName]['branches'][$branchId]);

263 private function getEmptyLinesForFile(string $filename): array
264 {
265 if (!isset(self::$emptyLineCache[$filename])) {
266 self::$emptyLineCache[$filename] = [];
267
268 if (is_file($filename)) {
269 $sourceLines = explode("\n", file_get_contents($filename));
270
271 foreach ($sourceLines as $line => $source) {
272 if (trim($source) === '') {
273 self::$emptyLineCache[$filename][] = ($line + 1);
274 }
275 }
276 }
277 }
278
279 return self::$emptyLineCache[$filename];
