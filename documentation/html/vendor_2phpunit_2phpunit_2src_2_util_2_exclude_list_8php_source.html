147 public static function addDirectory(string $directory): void
148 {

166 public function isExcluded(string $file): bool
167 {
168 if (defined('PHPUNIT_TESTSUITE')) {
169 return false;
170 }
171
172 self::initialize();
173
174 foreach (self::$directories as $directory) {
175 if (str_starts_with($file, $directory)) {
176 return true;
177 }
178 }
179
180 return false;
181 }
182
183 private static function initialize(): void
184 {
185 if (self::$initialized) {
186 return;
187 }
188
189 foreach (self::EXCLUDED_CLASS_NAMES as $className => $parent) {
190 if (!class_exists($className)) {
191 continue;
192 }
193
194 $directory = (new ReflectionClass($className))->getFileName();
195
196 for ($i = 0; $i < $parent; $i++) {
197 $directory = dirname($directory);
198 }
199
200 self::$directories[] = $directory;
201 }
202
203 // Hide process isolation workaround on Windows.
204 if (DIRECTORY_SEPARATOR === '\\') {
205 // tempnam() prefix is limited to first 3 chars.
206 // @see https://php.net/manual/en/function.tempnam.php
207 self::$directories[] = sys_get_temp_dir() . '\\PHP';
208 }
209
210 self::$initialized = true;
211 }
