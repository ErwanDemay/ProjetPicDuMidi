1019 private function telemetryInfo(): Telemetry\Info
1020 {
1021 $current = $this->system->snapshot();
1022
1023 $info = new Telemetry\Info(
1024 $current,
1025 $current->time()->duration($this->startSnapshot->time()),
1026 $current->memoryUsage()->diff($this->startSnapshot->memoryUsage()),
1027 $current->time()->duration($this->previousSnapshot->time()),
1028 $current->memoryUsage()->diff($this->previousSnapshot->memoryUsage()),
1029 );
1030
1031 $this->previousSnapshot = $current;
1032
1033 return $info;
1034 }
