127 private function analyse(string $filename): void
128 {
129 if (isset($this->classes[$filename])) {
130 return;
131 }
132
133 $source = file_get_contents($filename);
134 $linesOfCode = max(substr_count($source, "\n") + 1, substr_count($source, "\r") + 1);
135
136 if ($linesOfCode === 0 && !empty($source)) {
137 $linesOfCode = 1;
138 }
139
140 assert($linesOfCode > 0);
141
142 $parser = (new ParserFactory)->createForHostVersion();
143
144 try {
145 $nodes = $parser->parse($source);
146
147 assert($nodes !== null);
148
149 $traverser = new NodeTraverser;
150 $codeUnitFindingVisitor = new CodeUnitFindingVisitor;
151 $lineCountingVisitor = new LineCountingVisitor($linesOfCode);
152 $ignoredLinesFindingVisitor = new IgnoredLinesFindingVisitor($this->useAnnotationsForIgnoringCode, $this->ignoreDeprecatedCode);
153 $executableLinesFindingVisitor = new ExecutableLinesFindingVisitor($source);
154
155 $traverser->addVisitor(new NameResolver);
156 $traverser->addVisitor(new ParentConnectingVisitor);
157 $traverser->addVisitor($codeUnitFindingVisitor);
158 $traverser->addVisitor($lineCountingVisitor);

160 $traverser->addVisitor($executableLinesFindingVisitor);

162 /* @noinspection UnusedFunctionResultInspection */
163 $traverser->traverse($nodes);
164 // @codeCoverageIgnoreStart
165 } catch (Error $error) {
166 throw new ParserException(
167 sprintf(
168 'Cannot parse %s: %s',
169 $filename,
170 $error->getMessage(),

172 $error->getCode(),
173 $error,
174 );

204 private function findLinesIgnoredByLineBasedAnnotations(string $filename, string $source, bool $useAnnotationsForIgnoringCode): void
205 {
206 if (!$useAnnotationsForIgnoringCode) {
207 return;
208 }
209
210 $start = false;
211
212 foreach (token_get_all($source) as $token) {
213 if (!is_array($token) ||
214 !(T_COMMENT === $token[0] || T_DOC_COMMENT === $token[0])) {
215 continue;
216 }
217
218 $comment = trim($token[1]);
219
220 if ($comment === '// @codeCoverageIgnore' ||
221 $comment === '//@codeCoverageIgnore') {
222 $this->ignoredLines[$filename][] = $token[2];
223
224 continue;
225 }
226
227 if ($comment === '// @codeCoverageIgnoreStart' ||
228 $comment === '//@codeCoverageIgnoreStart') {
229 $start = $token[2];
230
231 continue;
232 }
233
234 if ($comment === '// @codeCoverageIgnoreEnd' ||
235 $comment === '//@codeCoverageIgnoreEnd') {
236 if (false === $start) {
237 $start = $token[2];
238 }
239
240 $this->ignoredLines[$filename] = array_merge(
241 $this->ignoredLines[$filename],
242 range($start, $token[2]),
243 );
244 }
245 }
