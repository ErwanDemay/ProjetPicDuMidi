64 public function testSuiteStarted(TestSuiteStarted $event): void
65 {
66 $testSuite = $event->testSuite();
67
68 if (!$this->isSummaryTestCountPrinted) {
69 $this->isSummaryTestCountPrinted = true;
70
71 $this->writeMessage(
72 'testCount',

75 }
76
77 if ($testSuite->isWithName() && $testSuite->name() === '') {
78 return;

80
81 $parameters = ['name' => $testSuite->name()];
82
83 if ($testSuite->isForTestClass()) {
84 assert($testSuite instanceof TestSuiteForTestClass);
85
86 $parameters['locationHint'] = sprintf(
87 'php_qn://%s::\\%s',
88 $testSuite->file(),
89 $testSuite->name()
90 );
91 } elseif ($testSuite->isForTestMethodWithDataProvider()) {
92 assert($testSuite instanceof TestSuiteForTestMethodWithDataProvider);
93
94 $parameters['locationHint'] = sprintf(
95 'php_qn://%s::\\%s',
96 $testSuite->file(),
97 $testSuite->name()
98 );
99
100 $parameters['name'] = $testSuite->methodName();
101 }
102
103 $this->writeMessage('testSuiteStarted', $parameters);

301 private function writeMessage(string $eventName, array $parameters = []): void
302 {
303 $this->printer->print(

348 private function message(Throwable $throwable): string
349 {
350 if (is_a($throwable->className(), FrameworkException::class, true)) {
351 return $throwable->message();
352 }
353
354 $buffer = $throwable->className();
355
356 if (!empty($throwable->message())) {
357 $buffer .= ': ' . $throwable->message();
358 }
359
360 return $buffer;
361 }

363 private function details(Throwable $throwable): string
364 {
365 $buffer = $throwable->stackTrace();
366
367 while ($throwable->hasPrevious()) {
368 $throwable = $throwable->previous();
369
370 $buffer .= sprintf(
371 "\nCaused by\n%s\n%s",
372 $throwable->description(),
373 $throwable->stackTrace()
374 );
375 }
376
377 return $buffer;
