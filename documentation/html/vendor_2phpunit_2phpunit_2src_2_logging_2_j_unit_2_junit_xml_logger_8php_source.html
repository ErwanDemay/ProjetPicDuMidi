360 private function name(Test $test): string
361 {
362 if ($test->isPhpt()) {
363 return basename($test->file());
364 }
365
366 assert($test instanceof TestMethod);
367
368 if (!$test->testData()->hasDataFromDataProvider()) {
369 return $test->methodName();
370 }
371
372 $dataSetName = $test->testData()->dataFromDataProvider()->dataSetName();
373
374 if (is_int($dataSetName)) {
375 return sprintf(
376 '%s with data set #%d',
377 $test->methodName(),
378 $dataSetName
379 );
380 }
381
382 return sprintf(
383 '%s with data set "%s"',
384 $test->methodName(),

386 );
