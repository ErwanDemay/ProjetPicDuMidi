40 public function linesToBeCovered(string $className, string $methodName): array|false
41 {
42 if (!$this->shouldCodeCoverageBeCollectedFor($className, $methodName)) {
43 return false;
44 }
45
46 $metadataForClass = Registry::parser()->forClass($className);
47 $classShortcut = null;
48
49 if ($metadataForClass->isCoversDefaultClass()->isNotEmpty()) {
50 if (count($metadataForClass->isCoversDefaultClass()) > 1) {
51 throw new CodeCoverageException(
52 sprintf(
53 'More than one @coversDefaultClass annotation for class or interface "%s"',
54 $className
55 )
56 );
57 }
58
59 $classShortcut = $metadataForClass->isCoversDefaultClass()->asArray()[0]->className();
60 }
61
62 $codeUnits = CodeUnitCollection::fromList();
63 $mapper = new Mapper;
64
65 foreach (Registry::parser()->forClassAndMethod($className, $methodName) as $metadata) {
66 if ($metadata->isCoversClass() || $metadata->isCoversFunction()) {
67 assert($metadata instanceof CoversClass || $metadata instanceof CoversFunction);
68
69 try {
70 $codeUnits = $codeUnits->mergeWith(
71 $mapper->stringToCodeUnits($metadata->asStringForCodeUnitMapper())
72 );
73 } catch (InvalidCodeUnitException $e) {
74 if ($metadata->isCoversClass()) {
75 $type = 'Class';
76 } else {
77 $type = 'Function';
78 }
79
80 throw new InvalidCoversTargetException(
81 sprintf(
82 '%s "%s" is not a valid target for code coverage',
83 $type,
84 $metadata->asStringForCodeUnitMapper()
85 ),
86 $e->getCode(),
87 $e
88 );
89 }
90 } elseif ($metadata->isCovers()) {
91 assert($metadata instanceof Covers);
92
93 $target = $metadata->target();
94
95 if (interface_exists($target)) {
96 throw new InvalidCoversTargetException(
97 sprintf(
98 'Trying to @cover interface "%s".',
99 $target
100 )
101 );
102 }
103
104 if ($classShortcut !== null && str_starts_with($target, '::')) {
105 $target = $classShortcut . $target;
106 }
107
108 try {
109 $codeUnits = $codeUnits->mergeWith($mapper->stringToCodeUnits($target));
110 } catch (InvalidCodeUnitException $e) {
111 throw new InvalidCoversTargetException(
112 sprintf(
113 '"@covers %s" is invalid',
114 $target
115 ),
116 $e->getCode(),
117 $e
118 );
119 }
120 }
121 }
122
123 return $mapper->codeUnitsToSourceLines($codeUnits);
124 }

207 public function shouldCodeCoverageBeCollectedFor(string $className, string $methodName): bool
208 {
209 $metadataForClass = Registry::parser()->forClass($className);
210 $metadataForMethod = Registry::parser()->forMethod($className, $methodName);
211
212 if ($metadataForMethod->isCoversNothing()->isNotEmpty()) {
213 return false;
214 }
215
216 if ($metadataForMethod->isCovers()->isNotEmpty() ||
217 $metadataForMethod->isCoversClass()->isNotEmpty() ||
218 $metadataForMethod->isCoversFunction()->isNotEmpty()) {
219 return true;
220 }
221
222 if ($metadataForClass->isCoversNothing()->isNotEmpty()) {
223 return false;
224 }
225
226 return true;
227 }
